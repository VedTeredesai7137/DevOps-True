# ✅ CORRECT S3 TERRAFORM CODE

Use this:

```hcl
terraform {
  required_providers {
    aws = {
      source  = "hashicorp/aws"
      version = "~> 3.27"
    }
  }
  required_version = ">= 0.14.9"
}

provider "aws" {
  region = "ap-south-1"
}

# S3 Bucket
resource "aws_s3_bucket" "ved_bucket" {
  bucket = "ved-devops8-unique-bucket-12345"  # MUST be globally unique

  tags = {
    Name        = "ved-devops8-main-bucket"
    Environment = "Dev"
  }
}

# Versioning
resource "aws_s3_bucket_versioning" "versioning" {
  bucket = aws_s3_bucket.ved_bucket.id

  versioning_configuration {
    status = "Enabled"
  }
}

# ACL (private)
resource "aws_s3_bucket_acl" "acl" {
  bucket = aws_s3_bucket.ved_bucket.id
  acl    = "private"
}
```
